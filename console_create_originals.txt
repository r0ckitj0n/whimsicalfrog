console.log('üöÄ Starting Original Maps Creation...');

// Landing Page Original Map (1 area)
const landingCoordinates = [
    { selector: 'area-1', top: 414, left: 466, width: 285, height: 153 }
];

// Main Room Original Map (5 areas)
const mainRoomCoordinates = [
    { selector: 'area-1', top: 243, left: 30, width: 234, height: 233 },
    { selector: 'area-2', top: 403, left: 390, width: 202, height: 241 },
    { selector: 'area-3', top: 271, left: 753, width: 170, height: 235 },
    { selector: 'area-4', top: 291, left: 1001, width: 197, height: 255 },
    { selector: 'area-5', top: 157, left: 486, width: 190, height: 230 }
];

async function createOriginalMap(roomType, coordinates, mapName) {
    try {
        console.log(`Creating ${mapName} Original map...`);
        
        const response = await fetch('api/room_maps.php', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify({
                action: 'save',
                room_type: roomType,
                map_name: 'Original',
                coordinates: coordinates
            })
        });

        const data = await response.json();
        
        if (data.success) {
            console.log(`‚úÖ ${mapName} Original map created successfully!`);
            console.log(`   Room: ${roomType}`);
            console.log(`   Areas: ${coordinates.length}`);
            return true;
        } else {
            console.error(`‚ùå Failed to create ${mapName} Original:`, data.message);
            return false;
        }
    } catch (error) {
        console.error(`‚ùå Error creating ${mapName} Original:`, error);
        return false;
    }
}

// Create both Original maps
async function createBothOriginals() {
    console.log('\nüìç Creating Landing Page Original...');
    const landingSuccess = await createOriginalMap('landing', landingCoordinates, 'Landing Page');
    
    console.log('\nüè† Creating Main Room Original...');
    const mainSuccess = await createOriginalMap('room_main', mainRoomCoordinates, 'Main Room');
    
    console.log('\nüìä RESULTS:');
    console.log(`Landing Page: ${landingSuccess ? '‚úÖ SUCCESS' : '‚ùå FAILED'}`);
    console.log(`Main Room: ${mainSuccess ? '‚úÖ SUCCESS' : '‚ùå FAILED'}`);
    
    if (landingSuccess && mainSuccess) {
        console.log('\nüéâ Both Original maps created successfully!');
        console.log('üí° Refresh the Room Mapper to see the new Original maps.');
    } else {
        console.log('\n‚ö†Ô∏è Some maps may not have been created. Check the errors above.');
    }
}

// Run the creation
createBothOriginals();
